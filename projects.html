Project 1 - Quadratic Equation Roots

Python program to find roots of a quadratic equation.

import math

def find_roots(a, b, c):
    discriminant = b**2 - 4*a*c
    if discriminant > 0:
        root1 = (-b + math.sqrt(discriminant)) / (2 * a)
        root2 = (-b - math.sqrt(discriminant)) / (2 * a)
        return f"Roots are real and distinct: {root1}, {root2}"
    elif discriminant == 0:
        root = -b / (2 * a)
        return f"Roots are real and equal: {root}"
    else:
        real_part = -b / (2 * a)
        imaginary_part = math.sqrt(-discriminant) / (2 * a)
        return f"Roots are complex: {real_part} ï¿½ {imaginary_part}i"

# Example:
a, b, c = 1, -3, 2
print(find_roots(a, b, c))

Project 2 - Prime Numbers from 1 to 100

Python program to print all prime numbers between 1 and 100.

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

primes = [i for i in range(1, 101) if is_prime(i)]
print("Prime numbers between 1 and 100:", primes)

Project 3 - Functions in Python

Python program demonstrating function as object, lambda, inner functions, and anonymous functions.

# Function as object
def greet():
    return "Hello, world!"

say_hello = greet
print(say_hello())

# Lambda function (arrow function equivalent)
add = lambda a, b: a + b
print(f"Sum using lambda: {add(5, 7)}")

# Inner function
def outer_function(msg):
    def inner_function():
        print(f"Message from inner function: {msg}")
    inner_function()

outer_function("Hello from the inner function!")
